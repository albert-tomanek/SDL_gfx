02546 
02547         /*
02548         * Check visibility of clipping rectangle
02549         */

02551                 return(0);
02552         }
02553 
02554         /*
02555         * Clip line and test if we have to draw 
02556         */
02557         if (!(_clipLine(dst, &x1, &y1, &x2, &y2))) {
02558                 return (0);
02559         }
02560 
02561         /*
02562         * Keep on working with 32bit numbers 
02563         */
02564         xx0 = x1;
02565         yy0 = y1;
02566         xx1 = x2;
02567         yy1 = y2;
02568 
02569         /*
02570         * Reorder points if required 
02571         */
02572         if (yy0 > yy1) {
02573                 tmp = yy0;
02574                 yy0 = yy1;
02575                 yy1 = tmp;
02576                 tmp = xx0;
02577                 xx0 = xx1;

02579         }
02580 
02581         /*
02582         * Calculate distance 
02583         */
02584         dx = xx1 - xx0;
02585         dy = yy1 - yy0;
02586 
02587         /*
02588         * Check for special cases 
02589         */

02591                 /*
02592                 * Vertical line 
02593                 */

02596                         return (vlineColor(dst, x1, y1, y2, color));

02601                                 return (pixelColor(dst, x1, y1, color));

02713                 * that we won't overrun the endpoint along the X axis. 
02714                 */
02715                 /*

02719

02733                                 y0p1++;
02734                         }

02738                         * weighting for this pixel, and the complement of the weighting for
02739                         * the paired pixel.

02741                         wgt = (erracc >> intshift) & 255;
02742                         result |= pixelColorWeightNolock (dst, xx0, yy0, color, 255 - wgt);
02743                         result |= pixelColorWeightNolock (dst, xx0, y0p1, color, wgt);
02744                 }
02745         }
02746 
02747         /*
02748         * Do we have to draw the endpoint 
02749         */
02750         if (draw_endpoint) {
02751                 /*

02753                 * need to be weighted. 
02754                 */
02755                 result |= pixelColorNolock (dst, x2, y2, color);
02756         }
02757 
02758         /* Unlock surface */
02759         if (SDL_MUSTLOCK(dst)) {
02760                 SDL_UnlockSurface(dst);
02761         }
02762 
02763         return (result);
02764 }

04513

04515         * Check visibility of clipping rectangle
04516         */

04518                 return(0);
04519         }
04520

04522         * Sanity check radii 
04523         */
04524         if (rad < 0) {
04525                 return (-1);
04526         }
04527

04529         * Fixup angles
04530         */

04533 
04534         /*
04535         * Special case for rad=0 - draw a point 
04536         */
04537         if (rad == 0) {
04538                 return (pixelColor(dst, x, y, color));
04539         }

04541         /*

05393         int texture_x_walker;    
05394         int texture_y_start;

05396         int pixels_written,write_width;
05397 
05398         /*
05399         * Check visibility of clipping rectangle
05400         */
05401         if ((dst->clip_rect.w==0) || (dst->clip_rect.h==0)) {
05402                 return(0);
05403         }
05404 
05405         /*
05406         * Swap x1, x2 if required to ensure x1<=x2
05407         */
05408         if (x1 > x2) {
05409                 xtmp = x1;
05410                 x1 = x2;
05411                 x2 = xtmp;
05412         }

05415         * Get clipping boundary and

05417         */

05420                 return(0);

05430         }
05431

05433         * Clip x 
05434         */
05435         if (x1 < left) {

05438         if (x2 > right) {
05439                 x2 = right;
05440         }
05441

05445         w = x2 - x1 + 1;
05446

05452                 texture_x_walker = texture->w + texture_x_walker ;

06112 double _evaluateBezier (double *data, int ndata, double t) 
06113 {
06114         double mu, result;
06115         int n,k,kn,nn,nkn;
06116         double blend,muk,munk;
06117 
06118         /* Sanity check bounds */
06119         if (t<0.0) {
06120                 return(data[0]);
06121         }
06122         if (t>=(double)ndata) {
06123                 return(data[ndata-1]);
06124         }
06125 
06126         /* Adjust t to the range 0.0 to 1.0 */ 
06127         mu=t/(double)ndata;
06128 
06129         /* Calculate interpolate */
06130         n=ndata-1;
06131         result=0.0;
06132         muk = 1;
06133         munk = pow(1-mu,(double)n);
06134         for (k=0;k<=n;k++) {
06135                 nn = n;
06136                 kn = k;
06137                 nkn = n - k;
06138                 blend = muk * munk;
06139                 muk *= mu;
06140                 munk /= (1-mu);
06141                 while (nn >= 1) {
06142                         blend *= nn;
06143                         nn--;
06144                         if (kn > 1) {
06145                                 blend /= (double)kn;
06146                                 kn--;
06147                         }
06148                         if (nkn > 1) {
06149                                 blend /= (double)nkn;
06150                                 nkn--;
06151                         }
06152                 }
06153                 result += data[k] * blend;
06154         }
06155 
06156         return (result);
06157 }
